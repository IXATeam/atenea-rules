var xml = new System.Text.StringBuilder();
xml.AppendLine("<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>");
xml.AppendLine("<worksheet xmlns=\"http://schemas.openxmlformats.org/spreadsheetml/2006/main\">");
xml.AppendLine("<sheetData>");
int rowIdx = 1;
foreach (var row in rows) {
    xml.Append("<row r=\"" + rowIdx + "\">");
    int colIdx = 1;
    foreach (var cell in row) {
        string colLetter = GetColLetter(colIdx);
        string cellRef = colLetter + rowIdx;
        // Escapar caracteres especiales XML
        string val = cell.Replace("&","&amp;").Replace("<","&lt;").Replace(">","&gt;").Replace("\"","&quot;");
        xml.Append("<c r=\"" + cellRef + "\" t=\"inlineStr\"><is><t>" + val + "</t></is></c>");
        colIdx++;
    }
    xml.AppendLine("</row>");
    rowIdx++;
}
xml.AppendLine("</sheetData></worksheet>");
